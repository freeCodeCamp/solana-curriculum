  // All rows
  board.forEach(r => {

    if (rowSum === playerOne * 3 || rowSum === playerTwo * 3) {
      return true;
    }
  });
  // All columns
  for (let i = 0; i < board.length; i++) {
    const columnSum = board[0][i] + board[1][i] + board[2][i];
